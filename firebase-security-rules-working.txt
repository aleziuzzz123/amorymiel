rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow all reads and writes for authenticated users (TEMPORARY FOR TESTING)
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
